model Pulses "example pulse flow generator" import Physiolibrary.Types.*; Physiolibrary.Types.RealIO.VolumeFlowRateOutput volumeflowrate; discrete Time T0 "beginning of cardiac cycle"; Boolean b(start=false); discrete Time HP "duration of cardiac cycle"; parameter Frequency HR=1.2; Time tc "relative time in carciac cycle"; parameter Time TD1=0.07 "relative time of start of systole"; discrete Time TD2 "relative time of end of systole"; parameter VolumeFlowRate QP=0.000424 "peak volume flowrate"; equation b = time - pre(T0) >= pre(HP) "true if new cardiac cycle begins"; when {initial(),b} then T0 = time "set beginning of cardiac cycle"; HP = 1/HR "update length of carciac cycle"; TD2 = TD1 + (2/5)*HP "compute end time of systole"; end when; tc = time - T0 "relative time in carciac cycle"; volumeflowrate = if tc < TD1 then 0 else if tc < TD2 then sin((tc - TD1)/ (TD2 - TD1)*Modelica.Constants.pi)*QP else 0 "zero before and after systole, otherwise sin up to peak flow"; end Pulses;
